## Eine Brute Force Lösung
``` c++
#include <cstdint>
#include <iostream>

int main(){
 int64_t input;
 cin << input;
 
 int64_t line = 0;
 int64_t linesize = 1;
 int64_t i = 0;
 while(i < input){
  i += linesize;
  line++;
  linesize += 2;
 }
 
 int64_t line_middle = line*line - line + 1;
 int64_t column = input - line_middle;
 column = abs(column);
 line--;
 int64_t result = line+column;
 cout << result << endl;
 return 0;
}

```
